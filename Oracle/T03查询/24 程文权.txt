--　5、列出部门名称和这些部门的员工信息，包括那些没有员工的部门。
    select d.dname , e.*
    from dept d 
    left join emp e 
    on d.deptno = e.deptno

--　6、列出所有job为“CLERK”（办事员）的姓名及其部门名称。
    select e.ename , d.dname
    from emp e 
    left join dept d
    on e.deptno = d.deptno
    where e.job = 'CLERK'
